using Azure.Messaging.EventHubs;
using Azure.Messaging.EventHubs.Consumer;
using Microsoft.Data.SqlClient;
using System.Text.Json;

public class CesConsumer
{
    private readonly string _eventHubConn;
    private readonly string _eventHubName;
    private readonly string _sqlConn;

    public CesConsumer(string eventHubConn, string eventHubName, string sqlConn)
    {
        _eventHubConn = eventHubConn;
        _eventHubName = eventHubName;
        _sqlConn = sqlConn;
    }

    public async Task RunAsync(CancellationToken ct)
    {
        var consumer = new EventHubConsumerClient(
            EventHubConsumerClient.DefaultConsumerGroupName,
            _eventHubConn,
            _eventHubName);

        await foreach (PartitionEvent evt in consumer.ReadEventsAsync(ct))
        {
            var json = Encoding.UTF8.GetString(evt.Data.Body.ToArray());
            var cloudEvent = JsonDocument.Parse(json).RootElement;

            var data = cloudEvent.GetProperty("data");
            var op = data.GetProperty("operation").GetString();
            var schema = data.GetProperty("schema").GetString();
            var table = data.GetProperty("table").GetString();

            await ApplyChangeAsync(schema, table, op, data);
        }
    }

    private async Task ApplyChangeAsync(string schema, string table, string op, JsonElement data)
    {
        using var conn = new SqlConnection(_sqlConn);
        await conn.OpenAsync();

        using var tx = conn.BeginTransaction();

        try
        {
            switch (op)
            {
                case "INSERT":
                    await ApplyInsertAsync(conn, tx, schema, table, data);
                    break;
                case "UPDATE":
                    await ApplyUpdateAsync(conn, tx, schema, table, data);
                    break;
                case "DELETE":
                    await ApplyDeleteAsync(conn, tx, schema, table, data);
                    break;
            }

            tx.Commit();
        }
        catch
        {
            tx.Rollback();
            throw;
        }
    }
}
